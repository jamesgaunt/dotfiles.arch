# Setup a VM for testing

# VM and UEFI firmware
sudo pacman -S --needed qemu-full edk2-ovmf

# does CPU support virtualization
grep -Ec 'vmx|svm' /proc/cpuinfo

# create virtual disc
qemu-img create -f qcow2 archtest.qcow2 30G

# create local copy of OVMF_VARS.4m.fd
cp /usr/share/edk2/x64/OVMF_VARS.4m.fd ./archtest_VARS.4m.fd

# confirm OVMF UEFI filename
ls /usr/share/edk2/x64/

# Launch the boot ISO
qemu-system-x86_64 \
  -enable-kvm \
  -m 4G \
  -cpu host \
  -smp 4 \
  -drive if=pflash,format=raw,readonly=on,file=/usr/share/edk2/x64/OVMF_CODE.4m.fd \
  -drive if=pflash,format=raw,file=./archtest_VARS.4m.fd \
  -drive file=archtest.qcow2,if=none,id=disk0,format=qcow2 \
  -device virtio-blk-pci,drive=disk0,serial=beast001 \
  -cdrom /home/james/Downloads/archlinux-2026.06.01-x86_64.iso \
  -nic user,hostfwd=tcp::2222-:22 \
  -boot menu=on

# SSH
ssh -p 2222 root@localhost
